        "# Login System with 3 Attempts and User Database\n",
        "# -------------------------------\n",
        "\n",
        "print(\"===== ACCOUNT SETUP =====\")\n",
        "user_database = {}\n",
        "\n",
        "# Step 1: Take number of users\n",
        "total_users = int(input(\"How many accounts do you want to create? \"))\n",
        "\n",
        "# Step 2: Register each user\n",
        "for i in range(total_users):\n",
        "    username = input(f\"Enter username for user {i + 1}: \")\n",
        "    password = input(f\"Enter password for {username}: \")\n",
        "    user_database[username] = password\n",
        "\n",
        "print(\"\\n===== LOGIN PANEL =====\")\n",
        "login_user = input(\"Enter your username: \")\n",
        "\n",
        "# Step 3: Verify username\n",
        "if login_user in user_database:\n",
        "    max_attempts = 3\n",
        "    for attempt in range(max_attempts):\n",
        "        login_pass = input(\"Enter your password: \")\n",
        "\n",
        "        if login_pass == user_database[login_user]:\n",
        "            print(f\"Login Successful! Welcome, {login_user}.\")\n",
        "            break\n",
        "        else:\n",
        "            print(f\"Incorrect password. Attempts left: {max_attempts - attempt - 1}\")\n",
        "\n",
        "        if attempt == max_attempts - 1:\n",
        "            print(\"Too many failed attempts. Access locked.\")\n",
        "else:\n",
        "    print(\"Username not found. Please create an account first.\")\n",
        "\n",
        "print(\"\\n===== USER DATABASE (For Admin View) =====\")\n",
